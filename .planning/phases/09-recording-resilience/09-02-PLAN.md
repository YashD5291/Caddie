<objective>
Detect CoreAudio device disconnection mid-recording and gracefully stop recording through the state machine.

Purpose: If the user unplugs headphones, disconnects a USB mic, or the meeting app's audio device goes away during recording, Caddie must stop gracefully rather than crash or produce a corrupted file. The device alive listener detects the disconnect, propagates it through AudioRecorder to RecordingCoordinator, which drives a clean stop via the state machine.

Output: End-to-end disconnect handling from CoreAudio listener to state machine transition, with the recording safely finalized.
</objective>

    ```swift
    private func registerDeviceAliveListener() {
        guard aggregateDeviceID != kAudioObjectUnknown else { return }

        var address = AudioObjectPropertyAddress(
            mSelector: kAudioDevicePropertyDeviceIsAlive,
            mScope: kAudioObjectPropertyScopeGlobal,
            mElement: kAudioObjectPropertyElementMain
        )

        let status = AudioObjectAddPropertyListener(
            aggregateDeviceID,
            &address,
            deviceAliveListener,
            Unmanaged.passUnretained(self).toOpaque()
        )
        if status != noErr {
            logger.warning("Failed to register device alive listener: OSStatus \(status)")
        }
    }
    ```

    3. Add a free function (outside the class, like the render callback pattern) for the listener:
    ```swift
    private func deviceAliveListener(
        objectID: AudioObjectID,
        numberAddresses: UInt32,
        addresses: UnsafePointer<AudioObjectPropertyAddress>,
        clientData: UnsafeMutableRawPointer?
    ) -> OSStatus {
        guard let clientData else { return noErr }
        let capture = Unmanaged<SystemAudioCapture>.fromOpaque(clientData).takeUnretainedValue()

        // Check if device is still alive
        var isAlive: UInt32 = 1
        var size = UInt32(MemoryLayout<UInt32>.size)
        var address = AudioObjectPropertyAddress(
            mSelector: kAudioDevicePropertyDeviceIsAlive,
            mScope: kAudioObjectPropertyScopeGlobal,
            mElement: kAudioObjectPropertyElementMain
        )
        AudioObjectGetPropertyData(objectID, &address, 0, nil, &size, &isAlive)

        if isAlive == 0 {
            capture.onDisconnect?()
        }

        return noErr
    }
    ```

    NOTE: Using `Unmanaged.passUnretained(self)` here is safe because:
    - The listener is removed in `stop()` and `cleanup()` BEFORE self can deallocate
    - Unlike the render callback (which may fire after stop returns), `AudioObjectRemovePropertyListener` is synchronous -- the listener won't fire after removal

    4. In `stop()` and `cleanup()`, remove the property listener BEFORE destroying the aggregate device:
    ```swift
    private func removeDeviceAliveListener() {
        guard aggregateDeviceID != kAudioObjectUnknown else { return }
        var address = AudioObjectPropertyAddress(
            mSelector: kAudioDevicePropertyDeviceIsAlive,
            mScope: kAudioObjectPropertyScopeGlobal,
            mElement: kAudioObjectPropertyElementMain
        )
        AudioObjectRemovePropertyListener(
            aggregateDeviceID,
            &address,
            deviceAliveListener,
            Unmanaged.passUnretained(self).toOpaque()
        )
    }
    ```

    Call `removeDeviceAliveListener()` at the start of `stop()` (before AudioOutputUnitStop) and in `cleanup()` (before destroyAggregateDevice).
