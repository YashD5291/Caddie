    ```swift
    /// Start capturing from a specific device directly (no process tap, no aggregate device).
    /// Used when user selects a Loopback virtual device or other input device.
    func start(deviceUID: String, onBuffer: @escaping BufferCallback) throws {
        guard !isRunning else {
            logger.warning("SystemAudioCapture already running")
            return
        }
        self.onBuffer = onBuffer

        guard let deviceID = AudioDevice.lookup(by: deviceUID)?.id else {
            throw CaptureError.deviceNotFound(deviceUID)
        }

        do {
            // Skip setupTap() entirely -- no tap, no aggregate device
            // Direct HAL AudioUnit on the resolved device
            try setupAudioUnitForDevice(deviceID)
            registerDeviceAliveListenerForDevice(deviceID)
            try startAudioUnit()

            directDeviceID = deviceID
            isRunning = true
            logger.info("System audio capture started via direct device (UID: \(deviceUID))")
        } catch {
            logger.error("Failed to start direct device capture: \(error.localizedDescription)")
            cleanupDirectDevice()
            throw error
        }
    }
    ```

    Add `setupAudioUnitForDevice(_ deviceID: AudioDeviceID)` -- same setup as existing `setupAudioUnit()` but uses the passed deviceID instead of `aggregateDeviceID`. Follow TN2091 order exactly:
    1. Find HAL Output component
    2. Create instance
    3. Enable input on bus 1
    4. Disable output on bus 0
    5. Set device via `kAudioOutputUnitProperty_CurrentDevice`
    6. Set 16kHz mono Int16 format on bus 1 output scope
    7. Create RenderContext, set render callback
    8. Initialize AudioUnit

    This method reuses the existing `systemAudioRenderCallback` free function (same callback works for both paths -- it uses RenderContext, not self).

    Add `registerDeviceAliveListenerForDevice(_ deviceID: AudioDeviceID)` -- same as existing `registerDeviceAliveListener()` but operates on the passed deviceID instead of `aggregateDeviceID`. Store the deviceID for later removal.

    Add `removeDeviceAliveListenerForDevice()` -- mirrors `removeDeviceAliveListener()` but for direct device.

    Add `cleanupDirectDevice()` -- cleanup for the direct device path (no aggregate/tap to destroy).

    Modify `stop()` to handle direct device path:
    ```swift
    func stop() {
        guard isRunning else { return }
        isRunning = false

        if directDeviceID != kAudioObjectUnknown {
            // Direct device path cleanup
            removeDeviceAliveListenerForDevice()
        } else {
            // Aggregate device path cleanup (existing)
            removeDeviceAliveListener()
        }

        if let unit = audioUnit {
            AudioOutputUnitStop(unit)
            AudioUnitUninitialize(unit)
            AudioComponentInstanceDispose(unit)
            audioUnit = nil
        }

        if let ctx = renderContext {
            ctx.audioUnit = nil
            ctx.onBuffer = nil
            Unmanaged.passUnretained(ctx).release()
            renderContext = nil
        }

        if directDeviceID != kAudioObjectUnknown {
            directDeviceID = kAudioObjectUnknown
        } else {
            destroyAggregateDevice()
            destroyTap()
        }

        onBuffer = nil
        logger.info("System audio capture stopped")
    }
    ```

    Also modify `cleanup()` similarly to handle both paths.
