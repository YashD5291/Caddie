<objective>
Add a HAL AudioUnit capture path to MicrophoneCapture for device-specific input.

Purpose: MicrophoneCapture currently uses AVAudioEngine which is hardwired to the system default mic. When a user selects a specific device (e.g., Loopback virtual device, USB mic), MicrophoneCapture needs to open that device directly via HAL AudioUnit -- the only CoreAudio API that supports targeting a specific AudioDeviceID for input.

Output: MicrophoneCapture with dual start paths -- `start(onBuffer:)` for v1.0 default behavior, `start(deviceUID:onBuffer:)` for HAL AudioUnit device capture.
</objective>

    ```swift
    /// Start capturing from a specific device via HAL AudioUnit.
    /// - Parameters:
    ///   - deviceUID: Persistent UID string of the target device.
    ///   - onBuffer: Callback receiving Int16 PCM samples at 16kHz mono.
    func start(deviceUID: String, onBuffer: @escaping BufferCallback) throws {
        guard !isRunning else {
            logger.warning("MicrophoneCapture already running")
            return
        }
        self.onBuffer = onBuffer

        // Resolve UID to transient AudioDeviceID
        guard let deviceID = AudioDevice.lookup(by: deviceUID)?.id else {
            throw CaptureError.deviceNotFound(deviceUID)
        }

        // Follow TN2091 order exactly: enable IO -> set device -> set format -> set callback -> init -> start
        try setupHALAudioUnit(deviceID: deviceID)
        try startHALAudioUnit()

        isUsingHAL = true
        isRunning = true
        logger.info("Microphone capture started via HAL AudioUnit (device: \(deviceUID))")
    }
    ```

    Implement `setupHALAudioUnit(deviceID:)` following TN2091 order (copy the pattern from SystemAudioCapture.setupAudioUnit() but use the passed deviceID instead of aggregateDeviceID):

    Step 1: Find HAL Output component via `AudioComponentDescription` with `kAudioUnitSubType_HALOutput`.
    Step 2: Create instance via `AudioComponentInstanceNew`.
    Step 3: Enable input on bus 1 via `kAudioOutputUnitProperty_EnableIO` on `kAudioUnitScope_Input` element 1.
    Step 4: Disable output on bus 0 via `kAudioOutputUnitProperty_EnableIO` on `kAudioUnitScope_Output` element 0.
    Step 5: Set device via `kAudioOutputUnitProperty_CurrentDevice` on `kAudioUnitScope_Global` element 0.
    Step 6: Set output format on bus 1 output scope to 16kHz mono Int16 via `kAudioUnitProperty_StreamFormat` on `kAudioUnitScope_Output` element 1.
    Step 7: Create RenderContext, set render callback via `kAudioOutputUnitProperty_SetInputCallback` on `kAudioUnitScope_Global` element 0. Use `Unmanaged.passRetained(context).toOpaque()` for inputProcRefCon.
    Step 8: Initialize via `AudioUnitInitialize`.

    Add a file-scope render callback function (same signature and logic as systemAudioRenderCallback in SystemAudioCapture.swift):
    ```swift
    private func microphoneRenderCallback(
        inRefCon: UnsafeMutableRawPointer,
        ioActionFlags: UnsafeMutablePointer<AudioUnitRenderActionFlags>,
        inTimeStamp: UnsafePointer<AudioTimeStamp>,
        inBusNumber: UInt32,
        inNumberFrames: UInt32,
        ioData: UnsafeMutablePointer<AudioBufferList>?
    ) -> OSStatus {
        let context = Unmanaged<MicrophoneCapture.RenderContext>.fromOpaque(inRefCon).takeUnretainedValue()
        guard let unit = context.audioUnit else { return noErr }

        let byteSize = Int(inNumberFrames) * MemoryLayout<Int16>.size
        let buffer = UnsafeMutablePointer<Int16>.allocate(capacity: Int(inNumberFrames))
        defer { buffer.deallocate() }

        var bufferList = AudioBufferList(
            mNumberBuffers: 1,
            mBuffers: AudioBuffer(
                mNumberChannels: 1,
                mDataByteSize: UInt32(byteSize),
                mData: buffer
            )
        )

        let status = AudioUnitRender(unit, ioActionFlags, inTimeStamp, 1, inNumberFrames, &bufferList)
        guard status == noErr else { return status }

        let sampleCount = Int(inNumberFrames)
        let bufferPtr = UnsafeBufferPointer(start: buffer, count: sampleCount)
        context.onBuffer?(bufferPtr, sampleCount)
        return noErr
    }
    ```

    ```swift
    func stop() {
        guard isRunning else { return }
        isRunning = false

        if isUsingHAL {
            // HAL path cleanup
            if let unit = halAudioUnit {
                AudioOutputUnitStop(unit)
                AudioUnitUninitialize(unit)
                AudioComponentInstanceDispose(unit)
                halAudioUnit = nil
            }
            if let ctx = renderContext {
                ctx.audioUnit = nil
                ctx.onBuffer = nil
                Unmanaged.passUnretained(ctx).release()
                renderContext = nil
            }
            isUsingHAL = false

            engine?.stop()
            engine = nil
            converter = nil
        }

        onBuffer = nil
        logger.info("Microphone capture stopped")
    }
    ```
